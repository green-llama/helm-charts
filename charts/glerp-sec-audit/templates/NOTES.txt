Trigger a Manual Scan
---------------------
  # kube-bench
  kubectl create job kube-bench-manual-$(date +%s) \
    --from=cronjob/{{ include "glerp-sec-audit.fullname" . }}-kube-bench \
    -n {{ .Release.Namespace }} --request-timeout=5s

  # OWASP ZAP (per target)
{{- range .Values.scanTargets }}
  kubectl create job zap-{{ .name }}-manual-$(date +%s) \
    --from=cronjob/{{ include "glerp-sec-audit.fullname" $ }}-zap-{{ .name | lower | replace "_" "-" }} \
    -n {{ $.Release.Namespace }} --request-timeout=5s
{{- end }}

Debug — Trivy Metrics
---------------------
  # Confirm metric name and label structure (status=Pass|Fail|Warn or PASS|FAIL|WARN)
  kubectl exec -n {{ .Release.Namespace }} \
    deploy/{{ include "glerp-sec-audit.fullname" . }}-trivy-operator -- \
    wget -qO- http://localhost:8080/metrics 2>/dev/null \
    | grep "^trivy_cluster_compliance" | head -10

  # Confirm vulnerability scan data exists (0 = VulnerabilityReports empty, delete to re-scan)
  kubectl exec -n {{ .Release.Namespace }} \
    deploy/{{ include "glerp-sec-audit.fullname" . }}-trivy-operator -- \
    wget -qO- http://localhost:8080/metrics 2>/dev/null \
    | grep "^trivy_image_vulnerabilities" | wc -l

  # Force re-scan when VulnerabilityReports are stale/empty
  kubectl delete vulnerabilityreports -A --all --request-timeout=30s

Debug — Pushgateway (kube-bench metrics)
-----------------------------------------
  # List what kube-bench metrics are stored and their instance labels
  kubectl exec -n {{ .Release.Namespace }} \
    deploy/{{ include "glerp-sec-audit.fullname" . }}-prometheus-pushgateway -- \
    wget -qO- http://localhost:9091/metrics 2>/dev/null | grep "^kube_bench"

  # Clear all kube-bench entries (removes duplicate-instance entries)
  kubectl exec -n {{ .Release.Namespace }} \
    deploy/{{ include "glerp-sec-audit.fullname" . }}-prometheus-pushgateway -- \
    wget -q -O /dev/null --method=DELETE http://localhost:9091/metrics/job/kube-bench
